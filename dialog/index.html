var btn1 = document.getElementById("btn1");
new dialog(".dialog1", {
	title: "警告",
	'buttons': {
		'确定': function(api) {
			api.close();
		}
	}
}, function(api) {
	btn1.onclick = function() {
		api.open();
		api.setContent("你好");
		console.log(api.getButtons());
		console.log(api.isOpen());
	}
})

调用方法：

new dialog(selector,options,callback(api));

options参数

参数 默认值 说明
content 空 对话框内容
title 空 对话框标题
opacity 0.5 对话框遮罩层透明度
autoOpen false 是否自动开启对话框
isModel true 是否为模式兑换
buttons 空 自定义按键

callback(api)参数

方法 说明
open() 打开对话框
close() 关闭对话框
resize() 对话框形状自动调整
setContent() 设置对话框内容
getButtons() 获取按键对象
isOpen() 获取对话框是否打开状态
